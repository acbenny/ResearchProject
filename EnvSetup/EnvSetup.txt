kubectl create namespace microservices
kubectl config set-context --current --namespace=microservices

helm repo add incubator https://kubernetes-charts-incubator.storage.googleapis.com
helm install orientdb incubator/orientdb --version 0.1.2 --namespace microservices --set rootPassword=orientDBadmin

helm repo add hashicorp https://helm.releases.hashicorp.com
helm install consul hashicorp/consul --set global.name=consul -f consul-helm-config.yaml -n microservices
