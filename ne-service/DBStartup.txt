CREATE CLASS NetworkElement EXTENDS V;
CREATE SEQUENCE neIdSeq TYPE ORDERED START 100;
CREATE PROPERTY NetworkElement.neId INTEGER (MANDATORY TRUE, NOTNULL TRUE, DEFAULT "sequence('neIdSeq').next()");
CREATE PROPERTY NetworkElement.model STRING (MANDATORY TRUE, NOTNULL TRUE);
CREATE INDEX NetworkElement.neId ON NetworkElement(neId) UNIQUE;

CREATE CLASS Port EXTENDS V;
CREATE PROPERTY Port.logicalPortName STRING (MANDATORY TRUE, NOTNULL TRUE);

CREATE CLASS Ports EXTENDS E;
CREATE PROPERTY Ports.out LINK NetworkElement;
CREATE PROPERTY Ports.in LINK Port;
CREATE PROPERTY Ports.portName STRING;

CREATE CLASS Tag EXTENDS V;
CREATE PROPERTY Tag.outerTag INTEGER;

CREATE CLASS Tags EXTENDS E;
CREATE PROPERTY Tags.out LINK Port;
CREATE PROPERTY Tags.in LINK Tag;
CREATE PROPERTY Tags.tagId INTEGER;

CREATE CLASS Order EXTENDS V;
CREATE PROPERTY Order.serviceId STRING (MANDATORY TRUE, NOTNULL TRUE);
CREATE PROPERTY Order.orderId LONG;

CREATE CLASS AllocatedTags EXTENDS E;
CREATE PROPERTY AllocatedTags.out LINK Order;
CREATE PROPERTY AllocatedTags.in LINK Tag;

CREATE CLASS Vpn EXTENDS V;
CREATE PROPERTY Vpn.vpnName STRING (MANDATORY TRUE, NOTNULL TRUE);
CREATE INDEX Vpn.vpnName ON Vpn(vpnName) UNIQUE;

CREATE CLASS Vrf EXTENDS V;
CREATE PROPERTY Vrf.vrfName STRING (MANDATORY TRUE, NOTNULL TRUE);

CREATE CLASS VpnVrfs EXTENDS E;
CREATE PROPERTY VpnVrfs.out LINK Vpn;
CREATE PROPERTY VpnVrfs.in LINK Vrf;

CREATE CLASS AllocatedVrfs EXTENDS E;
CREATE PROPERTY AllocatedVrfs.out LINK Order;
CREATE PROPERTY AllocatedVrfs.in LINK Vrf;
CREATE PROPERTY AllocatedVrfs.interfaceId INTEGER;

CREATE CLASS Vrfs EXTENDS E;
CREATE PROPERTY Vrfs.out LINK NetworkElement;
CREATE PROPERTY Vrfs.in LINK Vrf;

CREATE CLASS Filters EXTENDS E;
CREATE PROPERTY Filters.out LINK NetworkElement;
CREATE PROPERTY Filters.in LINK Order;
CREATE PROPERTY Filters.id INTEGER;
